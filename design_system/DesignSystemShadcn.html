Design Tokens
shadcn/ui uses HSL CSS variables — paste this :root block into your global CSS. Dark mode variables go under .dark.
/* VSee Clinic × shadcn/ui Tokens — v4.0 */ /* Paste into globals.css */ @layer base { :root { /* Core */ --background: 0 0% 100%; --foreground: 222 47% 11%; /* Surfaces */ --card: 0 0% 100%; --card-foreground: 222 47% 11%; --muted: 210 40% 96%; --muted-foreground: 215 16% 47%; --accent: 154 60% 94%; --accent-foreground: 158 82% 22%; /* Brand — VSee Green */ --primary: 158 82% 29%; /* #0D875C */ --primary-foreground: 0 0% 100%; /* Semantic */ --secondary: 210 40% 96%; --secondary-foreground: 222 47% 11%; --destructive: 0 84% 60%; --destructive-foreground: 0 0% 100%; /* Structural */ --border: 214 32% 91%; --input: 214 32% 91%; --ring: 158 82% 29%; /* Shape */ --radius: 0.5rem; /* 8px — adjust per brand */ } .dark { --background: 222 47% 7%; --foreground: 210 40% 98%; --primary: 158 60% 52%; /* ... remaining dark tokens */ } }
Tailwind Config (tailwind.config.ts)
import type { Config } from "tailwindcss" const config: Config = { theme: { extend: { colors: { background: "hsl(var(--background))", foreground: "hsl(var(--foreground))", primary: { DEFAULT: "hsl(var(--primary))", foreground: "hsl(var(--primary-foreground))", }, destructive: { DEFAULT: "hsl(var(--destructive))", foreground: "hsl(var(--destructive-foreground))", }, muted: { DEFAULT: "hsl(var(--muted))", foreground: "hsl(var(--muted-foreground))", }, accent: { DEFAULT: "hsl(var(--accent))", foreground: "hsl(var(--accent-foreground))", }, border: "hsl(var(--border))", input: "hsl(var(--input))", ring: "hsl(var(--ring))", }, borderRadius: { lg: "var(--radius)", md: "calc(var(--radius) - 2px)", sm: "calc(var(--radius) - 4px)", }, }, }, }
